export const StatusBadge = ({ status }) => {
  const colors = {
    stable: { bg: '#E8F8EF', text: '#07C160' },
    beta: { bg: '#FFF8E6', text: '#FFA726' },
    new: { bg: '#FFF5F0', text: '#FF5000' },
  };
  const c = colors[status] || colors.stable;
  return (
    <span style={{
      display: 'inline-block', padding: '2px 8px', borderRadius: 4,
      background: c.bg, color: c.text, fontSize: 11, fontWeight: 600, textTransform: 'uppercase',
    }}>
      {status}
    </span>
  );
};
