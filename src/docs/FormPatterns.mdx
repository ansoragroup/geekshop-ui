<SectionTitle>Controlled vs. Uncontrolled</SectionTitle>

All form components support both **controlled** (you manage state) and **uncontrolled** (component manages state) usage via the `useControllableState` hook.

<SubTitle>Uncontrolled (simplest)</SubTitle>

Just provide `defaultValue`. The component manages its own state internally:

```tsx
import { Input, Checkbox, Switch } from '@ansoragroup/ui'

function SimpleForm() {
  return (
    <form>
      {/* Component manages its own value */}
      <Input label="Email" defaultValue="" />
      <Checkbox defaultChecked={false}>Subscribe to newsletter</Checkbox>
      <Switch defaultChecked={true}>Dark mode</Switch>
    </form>
  )
}
```

<SubTitle>Controlled (full control)</SubTitle>

Provide `value` + `onChange`. You manage the state:

```tsx
import { useState } from 'react'
import { Input, Checkbox, Switch } from '@ansoragroup/ui'

function ControlledForm() {
  const [email, setEmail] = useState('')
  const [subscribe, setSubscribe] = useState(false)
  const [darkMode, setDarkMode] = useState(true)

  return (
    <form>
      <Input label="Email" value={email} onChange={(e) => setEmail(e.target.value)} />
      <Checkbox checked={subscribe} onChange={setSubscribe}>Subscribe</Checkbox>
      <Switch checked={darkMode} onChange={setDarkMode}>Dark mode</Switch>
    </form>
  )
}
```

<SubTitle>How useControllableState Works</SubTitle>

The hook powers all form components internally. It determines whether the component is controlled or uncontrolled based on whether `value` is provided:

```tsx
import { useControllableState } from '@ansoragroup/ui'

function MyComponent({ value, defaultValue, onChange }) {
  const [internalValue, setInternalValue] = useControllableState({
    value,                              // if provided, component is controlled
    defaultValue: defaultValue ?? '',   // fallback for uncontrolled
    onChange,                           // called on every change
  })

  // Use internalValue and setInternalValue everywhere
  // The hook handles both modes transparently
}
```

<SectionTitle>Form Field Anatomy</SectionTitle>

A complete form field consists of four parts:

```
┌─────────────────────────────────────┐
│  Label                              │   <- Associates with input via htmlFor/id
├─────────────────────────────────────┤
│  [ Input field                    ] │   <- The interactive control
├─────────────────────────────────────┤
│  Helper text or error message       │   <- Contextual guidance
└─────────────────────────────────────┘
```

The `Input` and `TextArea` components have built-in support for all four parts:

```tsx
<Input
  label="Email Address"           // label text (auto-generates htmlFor/id)
  placeholder="you@example.com"   // placeholder
  helperText="We'll never share"  // helper text below input
  error="Invalid email format"    // replaces helper text in error state
/>
```

When `error` is provided, the input border turns red and the error message replaces the helper text.

<SectionTitle>Validation Patterns</SectionTitle>

<SubTitle>Inline Validation (on blur)</SubTitle>

Validate each field as the user leaves it:

```tsx
import { useState } from 'react'
import { Input, Button } from '@ansoragroup/ui'

function LoginForm() {
  const [email, setEmail] = useState('')
  const [emailError, setEmailError] = useState('')

  const validateEmail = () => {
    if (!email) {
      setEmailError('Email is required')
    } else if (!email.includes('@')) {
      setEmailError('Please enter a valid email')
    } else {
      setEmailError('')
    }
  }

  return (
    <form>
      <Input
        label="Email"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
        onBlur={validateEmail}
        error={emailError}
      />
      <Button variant="primary" size="lg" fullWidth type="submit">
        Sign In
      </Button>
    </form>
  )
}
```

<SubTitle>On-Submit Validation</SubTitle>

Validate all fields when the form is submitted:

```tsx
import { useState } from 'react'
import { Input, Button } from '@ansoragroup/ui'

function RegisterForm() {
  const [fields, setFields] = useState({ email: '', password: '' })
  const [errors, setErrors] = useState<Record<string, string>>({})

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    const newErrors: Record<string, string> = {}

    if (!fields.email) newErrors.email = 'Email is required'
    if (!fields.password) newErrors.password = 'Password is required'
    if (fields.password.length < 8) newErrors.password = 'Minimum 8 characters'

    setErrors(newErrors)
    if (Object.keys(newErrors).length === 0) {
      // Submit form
    }
  }

  return (
    <form onSubmit={handleSubmit}>
      <Input
        label="Email"
        value={fields.email}
        onChange={(e) => setFields(f => ({ ...f, email: e.target.value }))}
        error={errors.email}
      />
      <Input
        label="Password"
        type="password"
        value={fields.password}
        onChange={(e) => setFields(f => ({ ...f, password: e.target.value }))}
        error={errors.password}
      />
      <Button variant="primary" size="lg" fullWidth type="submit">
        Register
      </Button>
    </form>
  )
}
```

<SubTitle>Review Form</SubTitle>

```tsx
import { useState } from 'react'
import { Rating, TextArea, Button } from '@ansoragroup/ui'

function ReviewForm() {
  const [rating, setRating] = useState(0)
  const [review, setReview] = useState('')
  const [ratingError, setRatingError] = useState('')

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    if (rating === 0) {
      setRatingError('Please select a rating')
      return
    }
    // Submit review
  }

  return (
    <form onSubmit={handleSubmit} style={{ display: 'flex', flexDirection: 'column', gap: 16 }}>
      <div>
        <div style={{ fontSize: 14, fontWeight: 600, marginBottom: 8 }}>Rate this product</div>
        <Rating
          value={rating}
          onChange={(v) => { setRating(v); setRatingError(''); }}
          interactive
          size="lg"
        />
        {ratingError && (
          <div style={{ fontSize: 12, color: '#FF3B30', marginTop: 4 }}>{ratingError}</div>
        )}
      </div>

      <TextArea
        label="Your Review"
        value={review}
        onChange={(e) => setReview(e.target.value)}
        placeholder="Write your review..."
        maxLength={500}
        showCount
        rows={4}
      />

      <Button
        variant="primary"
        size="lg"
        fullWidth
        type="submit"
        disabled={rating === 0}
      >
        Submit Review
      </Button>
    </form>
  )
}
```

<SectionTitle>Error State Handling</SectionTitle>

Form components indicate errors through:

1. **Red border** -- Input/TextArea border color changes to `$color-error`
2. **Error message** -- Text below the field in red
3. **ARIA attributes** -- `aria-invalid="true"` and `aria-describedby` linking to the error message

```tsx
// Error state
<Input
  label="Email"
  value=""
  error="Email is required"        // Triggers red border + error text
/>

// Valid state (no error prop or empty string)
<Input
  label="Email"
  value="user@example.com"
  error=""                          // No error shown
/>
```

<SectionTitle>Disabled State Handling</SectionTitle>

All form components support a `disabled` prop:

```tsx
<Input label="Email" disabled value="locked@example.com" />
<Checkbox disabled checked>Accepted terms</Checkbox>
<Switch disabled checked>Feature flag</Switch>
<Radio disabled value="option1">Option 1</Radio>
<Button disabled>Submit</Button>
```

Disabled components:
- Reduce opacity to 0.5
- Show `cursor: not-allowed`
- Do not fire `onChange` or `onClick`
- Set `aria-disabled="true"` and the native `disabled` attribute

<SectionTitle>Third-Party Form Libraries</SectionTitle>

All form components spread `...rest` props onto their root or input element, making them compatible with form libraries:

<SubTitle>react-hook-form</SubTitle>

```tsx
import { useForm } from 'react-hook-form'
import { Input, Button } from '@ansoragroup/ui'

function HookForm() {
  const { register, handleSubmit, formState: { errors } } = useForm()

  return (
    <form onSubmit={handleSubmit((data) => console.log(data))}>
      <Input
        label="Email"
        {...register('email', { required: 'Email is required' })}
        error={errors.email?.message as string}
      />
      <Button variant="primary" type="submit">Submit</Button>
    </form>
  )
}
```

<SubTitle>Formik</SubTitle>

```tsx
import { Formik, Form } from 'formik'
import { Input, Button } from '@ansoragroup/ui'

function FormikForm() {
  return (
    <Formik
      initialValues={{ email: '' }}
      onSubmit={(values) => console.log(values)}
    >
      {({ values, errors, handleChange, handleBlur }) => (
        <Form>
          <Input
            label="Email"
            name="email"
            value={values.email}
            onChange={handleChange}
            onBlur={handleBlur}
            error={errors.email}
          />
          <Button variant="primary" type="submit">Submit</Button>
        </Form>
      )}
    </Formik>
  )
}
```

These integrations work because the components:
- Accept `name`, `value`, `onChange`, `onBlur` props
- Spread additional props (`data-*`, `aria-*`, `ref`) onto the underlying `<input>` element
- Use `forwardRef` for ref access to the native input
